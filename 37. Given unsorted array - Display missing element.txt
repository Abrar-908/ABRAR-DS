#include <stdio.h>

int main() {
    int arr[100], n, i;
    int sum = 0, expectedSum, missing;

    printf("Enter size of array (n-1 elements, numbers from 1 to n): ");
    scanf("%d", &n);

    printf("Enter %d elements:\n", n - 1);
    for (i = 0; i < n - 1; i++) {
        scanf("%d", &arr[i]);
        sum += arr[i];
    }

    // Formula for sum of first n natural numbers
    expectedSum = n * (n + 1) / 2;

    missing = expectedSum - sum;
    printf("Missing element is: %d\n", missing);

    return 0;
}
