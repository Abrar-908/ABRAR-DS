#include <stdio.h>
#define MAX 100

int visited[MAX];

// DFS function
void DFS(int graph[MAX][MAX], int vertices, int start) {
    printf("%d ", start);
    visited[start] = 1;

    for (int i = 0; i < vertices; i++) {
        if (graph[start][i] == 1 && !visited[i]) {
            DFS(graph, vertices, i);
        }
    }
}

// Main function
int main() {
    int vertices;
    printf("Enter number of vertices: ");
    scanf("%d", &vertices);

    int graph[MAX][MAX];
    printf("Enter adjacency matrix of the graph (%d x %d):\n", vertices, vertices);
    for (int i = 0; i < vertices; i++) {
        for (int j = 0; j < vertices; j++) {
            scanf("%d", &graph[i][j]);
        }
    }

    int start;
    printf("Enter starting vertex (0 to %d): ", vertices - 1);
    scanf("%d", &start);

    // mark all as unvisited
    for (int i = 0; i < vertices; i++)
        visited[i] = 0;

    printf("DFS Traversal starting from vertex %d: ", start);
    DFS(graph, vertices, start);
    printf("\n");

    return 0;
}
